df_with_counts.to_crs(epsg=3857, inplace=True)

fig, ax = plt.subplots(1, 1)
divider = make_axes_locatable(ax)
cax = divider.append_axes("right", size="5%", pad=0.1)

df_with_counts["coords"] = df_with_counts["geometry"].apply(lambda x: x.representative_point().coords[:])
df_with_counts["coords"] = [coords[0] for coords in df_with_counts["coords"]]

for idx, row in df_with_counts.iterrows():
    ax.annotate(row["zipcode"], xy=row["coords"], horizontalalignment='center')

ax = df_with_counts.plot(
    figsize=(18, 18), 
    alpha=0.5, 
    edgecolor='k', 
    column="Number Auctioned", # This column determines the color
    ax=ax,
    cax=cax,
    legend=True)
ctx.add_basemap(ax)
